                        <exclude>lib/log4j-*.jar</exclude>

                        <goals>
                            <goal>copy-dependencies</goal>
                        </goals>

                            <includeScope>compile</includeScope>

            <plugin>
                <groupId>org.apache.maven.plugins</groupId>
                <artifactId>maven-antrun-plugin</artifactId>
                <version>3.1.0</version>
                <executions>
                    <!-- Purge conflicting logging libraries from OMOD lib/ just before packaging -->
                    <execution>
                        <id>purge-conflicting-logs</id>
                        <!-- Run after copy-dependencies (prepare-package) and BEFORE jar is packaged -->
                        <phase>prepare-package</phase>
                        <configuration>
                            <target>
                                <delete>
                                    <fileset dir="${project.build.outputDirectory}/lib">
                                        <include name="slf4j-*.jar"/>
                                        <include name="jcl-over-slf4j*.jar"/>
                                        <include name="commons-logging*.jar"/>
                                        <include name="logback-*.jar"/>

                            <goal>run</goal>
                        </goals>
                    </execution>

                        <id>rename-jar-to-omod</id>
                        <phase>package</phase>

                            <target>
                                <move file="${project.build.directory}/${project.build.finalName}.jar"
